# Production Keystore Configuration
# Add these lines to android/gradle.properties after generating your keystore
# Replace the placeholder values with your actual keystore information

# Keystore file path (relative to android/app directory)
CASEROAPP_UPLOAD_STORE_FILE=keystores/caseroapp-release.keystore

# Keystore alias (the alias you used when generating the keystore)
CASEROAPP_UPLOAD_KEY_ALIAS=caseroapp

# Keystore password (the password you set for the keystore)
CASEROAPP_UPLOAD_STORE_PASSWORD=YOUR_KEYSTORE_PASSWORD_HERE

# Key password (the password you set for the key, often same as keystore password)
CASEROAPP_UPLOAD_KEY_PASSWORD=YOUR_KEY_PASSWORD_HERE

# SECURITY NOTE:
# - Never commit these passwords to version control
# - Add android/gradle.properties to your .gitignore if it contains passwords
# - Consider using environment variables for CI/CD pipelines
# - Keep backups of both the keystore file and passwords in secure locations
